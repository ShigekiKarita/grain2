33         // init P2P
34         foreach (i; 0 .. count)
35         {
36             cudaSetDevice(i);
37             foreach (j; 0 .. count)
38             {
39                 if (i == j) continue;

41                 checkCuda(cudaDeviceCanAccessPeer(&ok, i, j));
42                 if (!ok)
43                 {
44                     debug
45                     {
46                         printf("[grain.warn]: no GPU P2P: %d -> %d\n", i, j);
47                     }
48                     continue;
49                 }
50                 cudaDeviceEnablePeerAccess(j, 0);
51             }
52         }
53     }

60     }
