52 struct ClDevice
53 {
54     @nogc nothrow:
55 
56     static cl_platform_id[] platformIds;
57     static typeof(this)[] platforms;
58 
59     cl_platform_id platformId;
60     cl_device_id[] deviceIds;
61 
62     static this()
63     {
64         import core.memory : pureMalloc;
65         // init device
66         cl_uint numPlatform;
67         checkCl(clGetPlatformIDs(0, null, &numPlatform));
68         auto platformIdsPtr = cast(cl_platform_id*)
69             pureMalloc(cl_platform_id.sizeof * numPlatform);
70         platformIds = platformIdsPtr[0 .. numPlatform];
71         checkCl(clGetPlatformIDs(numPlatform, platformIdsPtr, null));
72 
73         auto platformsPtr = cast(typeof(this)*)
74             pureMalloc(typeof(this).sizeof * numPlatform);
75         platforms = platformsPtr[0 .. numPlatform];
76         platforms[] = typeof(this).init;
77         foreach (pid, ref p; platforms)
78         {
79             p.platformId = platformIds[pid];
80         }
81 
82         foreach (pid, ref p; platforms)
83         {
84             cl_device_id device_id = null;
85             cl_uint numDevice;
86             checkCl(clGetDeviceIDs(p.platformId, CL_DEVICE_TYPE_DEFAULT,
87                                    0, null, &numDevice));
88             auto deviceIdsPtr = cast(cl_device_id*)
89                 pureMalloc(cl_device_id.sizeof * numDevice);
90             checkCl(clGetDeviceIDs(p.platformId, CL_DEVICE_TYPE_DEFAULT,
91                                    numDevice, deviceIdsPtr, null));
92             p.deviceIds = deviceIdsPtr[0 .. numDevice];
93         }
94         printInfo();
95     }
96 
97     static cl_device_id get(cl_uint device, cl_uint platform=0)
98     {
99         return platforms[platform].deviceIds[device];
100     }
101 
102     static void printInfo()
103     {
104         import core.stdc.stdio : printf;
105         foreach (pid, p; platforms)
106         {
107             foreach (did, d; p.deviceIds)
108             {
109                 debug printf("platform id: %d, device id: %d\n", pid, did);
110                 printDeviceInfo(d);
111             }
112         }
113     }
114 
115     static ~this()
116     {
117         import core.memory : pureFree;
118         foreach (p; platforms)
119         {
120             pureFree(p.deviceIds.ptr);
121         }
122         pureFree(platformIds.ptr);
123         pureFree(platforms.ptr);
124     }
125 }
