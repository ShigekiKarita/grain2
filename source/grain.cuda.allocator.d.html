17     Standard allocator methods per the semantics defined above. The
18     $(D deallocate) method is $(D @system) because it
19     may move memory around, leaving dangling pointers in user code. Somewhat
20     paradoxically, $(D malloc) is $(D @safe) but that's only useful to safe
21     programs that can afford to leak memory allocated.
22     */
23     @trusted @nogc nothrow
24     void[] allocate()(size_t bytes)
25     {
26         import grain.dpp.cuda_runtime_api : cudaMallocHost;
27

31         checkCuda(cudaMallocHost(&p, bytes));
32         return p ? p[0 .. bytes] : null;
33     }

41         return true;
42     }

49 struct CuMallocator

51     Opt opt;
52     alias opt this;
53 
54     /// device indicator
55     enum deviceof = "cuda";
56     enum pinned = false;
57 
58     /**
59     Standard allocator methods per the semantics defined above. The
60     $(D deallocate) method is $(D @system) because it
61     may move memory around, leaving dangling pointers in user code. Somewhat
62     paradoxically, $(D malloc) is $(D @safe) but that's only useful to safe
63     programs that can afford to leak memory allocated.
64     */
65     @trusted @nogc nothrow
66     void[] allocate()(size_t bytes)
67     {
68         // import grain.dpp.cuda_driver : cuMemAlloc_v2, CUdeviceptr;
69         if (!bytes) return null;
70 
71         void* p;
72         cudaSetDevice(this.opt.deviceId);
73         checkCuda(cudaMalloc(&p, bytes));
74         return p ? p[0 .. bytes] : null;
75     }
76 
77     /// Ditto
78     @system @nogc nothrow
79     bool deallocate()(void[] b)
80     {
81         cudaSetDevice(this.opt.deviceId);
82         checkCuda(cudaFree(b.ptr));
83         return true;
84     }
85 
86     enum instance = CuMallocator();
87 }
