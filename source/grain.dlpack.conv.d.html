10 enum DLDataType dataTypeOf(T) = {
11     DLDataType ret;
12     ret.bits = T.sizeof * 8;
13     ret.lanes = 1;
14     static if (is(T : __vector(V[N]), V, size_t N))
15     {
16         ret = dataTypeOf!V;
17         ret.bits = V.sizeof * 8;
18         ret.lanes = N;
19     }
20     else static if (is(T == bfloat16))
21     {
22         ret.code = kDLBFloat;
23     }
24     else static if (isFloatingPoint!T)
25     {
26         ret.code = kDLFloat;
27     }
28     else static if (isUnsigned!T)
29     {
30         ret.code = kDLUInt;
31     }
32     else static if (isIntegral!T)
33     {
34         ret.code = kDLInt;
35     }
36     else
37     {
38         static assert(false, "cannot convert to DLDataType: " ~ T.stringof);
39     }
40     return ret;
41 }();

85 /** Convert AnyTensor to DLManagedTensor
86 
87     Returns:
88     The DLManagedTensor to be consumed exactly once (i.e., call deleter once)
89 
90     See_also:
91     https://github.com/pytorch/pytorch/blob/v1.2.0/aten/src/ATen/DLConvertor.cpp
92 */
93 @nogc nothrow
94 DLManagedTensor* toDLPack(AnyTensor src)
95 {
96     import std.experimental.allocator : make, dispose;
97     import std.experimental.allocator.mallocator : Mallocator;
98 
99     auto manager = Mallocator.instance.make!DLManager;
100     manager.handle = src; // inc ref counts by copy
101     auto ret = &manager.tensor;
102     with (ret.dl_tensor)
103     {
104         ndim = cast(int) src.shape.length;
105         byte_offset = src.offset;
106         data = src.buffer.payload.ptr;
107         shape = &(*src.shape)[0];
108         strides = &(*src.strides)[0];
109         ctx = src.context;
110         dtype = src.dataType;
111     }
112     ret.manager_ctx = manager;
113     ret.deleter = (DLManagedTensor* self) @trusted {
114         // dec ref count by dtor in DLManager.handle
115         Mallocator.instance.dispose(cast(DLManager*) self.manager_ctx);
116     };
117     return ret;
118 }

120 /// Load a new AnyTensor from DLManagedTensor
121 @nogc nothrow
122 AnyTensor toAny(DLManagedTensor* src)
123 {
124     import grain.rc : RC;
125     AnyTensor ret;
126     ret.buffer = RC!DLBuffer.create(src).castTo!AnyBuffer;
127     with (src.dl_tensor)
128     {
129         ret.offset = byte_offset;
130         ret.shape = RC!(Array!long).create(shape[0 .. ndim]);
131         ret.strides = RC!(Array!long).create(strides[0 .. ndim]);
132         ret.context = ctx;
133         ret.dataType = dtype;
134     }
135     return ret;
136 }
